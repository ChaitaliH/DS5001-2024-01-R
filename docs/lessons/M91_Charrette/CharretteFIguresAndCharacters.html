def get_p_fig_for_char(char_id):
    X = FTP.xs(char_id, level=1)
    return (X / X.sum()).sort_values('n', ascending=False).style.background_gradient()
def get_p_char_for_fig(fig_id):
    X = FTP.loc[fig_id]
    return (X / X.sum()).sort_values('n', ascending=False).style.background_gradient()
